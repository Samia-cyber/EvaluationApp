Comment créer un administrateur de base de données SQL Server Management Studio (SSMS) : 


-- Étape 1 : Récupérer l'Id de l'utilisateur et l'Id du rôle Administrateur:

SELECT Id, Email FROM AspNetUsers WHERE Email = 'samouna@gmail.com'
SELECT Id FROM AspNetRoles WHERE Name = 'Administrateur'

)

-- Étape 2 : Vérifier que l'utilisateur a bien le rôle Administrateur:

SELECT 
    u.Email,
    r.Name as Role
FROM AspNetUsers u
JOIN AspNetUserRoles ur ON u.Id = ur.UserId
JOIN AspNetRoles r ON ur.RoleId = r.Id
WHERE u.Email = 'samouna@gmail.com'
